// fib.c

int fib(int n) {
    int i, n1 = 1, n2 = 1, tmp = 0;

    for (i = 2; i < n; ++i) {
        tmp = n1 + n2;
        n1 = n2;
        n2 = tmp;
    }

    return n2;
}
